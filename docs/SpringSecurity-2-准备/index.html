    # SwaggerConfig

    @Configuration
@EnableKnife4j
public class SwaggerConfig implements WebMvcConfigurer {

    /**
     * 创建docket对象
     *
     * @return {@link Docket}
     */
    @Bean
    public Docket docket() {
        return new Docket(DocumentationType.SWAGGER_2).select().apis(RequestHandlerSelectors.basePackage("com.simpleword")).build().apiInfo(apiInfo()).globalRequestParameters(globalRequestParameters());
    }

    /**
     * 添加全局请求参数token
     */
    private List<RequestParameter> globalRequestParameters() {
        return Collections.singletonList(new RequestParameterBuilder().name("token").description("用户token").in(ParameterType.HEADER).build());
    }

    /**
     * 创建apiInfo对象
     *
     * @return {@link ApiInfo}
     */
    public ApiInfo apiInfo() {
        return new ApiInfoBuilder().title("测试文档").version("1.0").description("测试使用文档").build();
    }
}

    # JwtUtil

    package com.simpleword.util;

import com.simpleword.properties.JwtProperties;
import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

import javax.annotation.PostConstruct;
import java.util.Date;
import java.util.Map;

@Component
public class JwtUtil {
    public static String key;

    public static Integer expiration;

    @Autowired
    private JwtProperties jwtProperties;

    private JwtUtil() {
    }

    /**
     * 创建jwt
     *
     * @param claims 载荷
     * @return {@link String}
     */
    public static String creatJwt(Map<String, Object> claims) {
        return Jwts.builder().signWith(SignatureAlgorithm.HS256, key)
                .setClaims(claims)
                .setExpiration(new Date(System.currentTimeMillis() + expiration * 60 * 60 * 1000))
                .compact();
    }

    /**
     * 解析 JWT
     *
     * @param token 令 牌
     * @return boolean
     */
    public static Claims parseJwt(String token) {
        return Jwts.parser().setSigningKey(key).parseClaimsJws(token).getBody();
    }


    /**
     * 获取用户 ID
     *
     * @param token 令 牌
     * @return {@link Long}
     */
    public static Long getUserId(String token) {
        return parseJwt(token).get("userId", Long.class);
    }


    /**
     * @param token 令 牌
     * @return {@link String}
     */
    public static String getUsername(String token) {
        return parseJwt(token).get("username", String.class);
    }

    @PostConstruct
    public void setExpiration() {
        JwtUtil.expiration = jwtProperties.getExpiration();
    }

    @PostConstruct
    public void setKey() {
        JwtUtil.key = jwtProperties.getKey();
    }
}

    # ResponseUtil

    //原生方式返回工具类
public class ResponseUtil {

    public static void out(HttpServletResponse response, Result r) {
      	//ObjectMapper由jackson提供用于序列化和反序列化json
        ObjectMapper mapper = new ObjectMapper();
      	//code 200
        response.setStatus(HttpStatus.OK.value());
      	//响应：json、UTF-8
        response.setContentType(MediaType.APPLICATION_JSON_UTF8_VALUE);
        try {
          	//将json写入响应体，r是待转换为json的对象
            mapper.writeValue(response.getWriter(), r);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

    public Result exception(Exception e) {
        return Result.build(201, "系统异常", null);
