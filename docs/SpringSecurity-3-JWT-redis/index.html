    # 自定义用户对象UserDetails

    Security自带的User往往不满足需求，自定义，但是至少要保证有用户名和密码两个属性

    //这里继承的User来自Security,可以更好的扩展
public class CustomUser extends User {
    //用户表自定义User
    private PerUser perUser;

    /**
     * 自定义用户
     *
     * @param perUser
     * @param authorities 权限
     */
    public CustomUser(PerUser perUser, Collection<? extends GrantedAuthority> authorities) {
        //这里是Security的user要求的属性，父类构造会在子类构造调用
        super(perUser.getUsername(), perUser.getPassword(), authorities);
        this.perUser = perUser;
    }

    public void setUser(PerUser perUser) {
        this.perUser = perUser;
    }
}

    # 自定义loadUserByUsername

    • 这里可以自定义类实现UserDetailsService接口，它只有一个方法
    • 此方法根据用户名返回用户对象
    • 第三个属性树用户的权限/角色
    @Component
public class UserDetailServiceImpl implements UserDetailsService {
    @Autowired
    private UserService userService;

    @Autowired
    private PermissionService permissionService;

    @Override
    public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
        //数据库查询根据用户名查用户
        LambdaQueryWrapper<PerUser> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(PerUser::getUsername, username);
        PerUser perUser = userService.getOne(wrapper);
        //简单校验
        if (perUser == null) {
            throw new UsernameNotFoundException("用户不存在");
        }
        if (perUser.getState() == 0) {
            throw new RuntimeException("用户已禁用");
        }
        //查询用户权限并封装到对象
        List<String> permissionList = permissionService.getPermissionByUsername(username);
        //指定类型集合,GrantedAuthority是接口，我们用实现类
        List<SimpleGrantedAuthority> authoritiesList = new ArrayList<>();
        // 遍历权限列表，将每个权限封装成SimpleGrantedAuthority对象
        for (String permission : permissionList) {
            authoritiesList.add(new SimpleGrantedAuthority(permission));
        }
        //返回User
        return new CustomUser(perUser, authoritiesList);
    }
}

    # 自定义加密机PasswordEncoder

    • 添加用户和登录都要用到加密，我们使用自定义的
    • 注意数据库密码要存储密文，加密算法与这里一致
    @Component
public class CustomMd5PasswordEncoder implements PasswordEncoder {
    //原始密码加密
    @Override
    public String encode(CharSequence rawPassword) {
        return DigestUtils.md5DigestAsHex(rawPassword.toString().getBytes());
    }

    //验证原始密码加密后与加密后的密码是否相同
    @Override
    public boolean matches(CharSequence rawPassword, String encodedPassword) {
        return encodedPassword.equals(DigestUtils.md5DigestAsHex(rawPassword.toString().getBytes()));
    }
}

    # 认证过滤器

    • 此类没有交给Spring管理，redis、manager通过构造注入，具体见Security配置类
    • 这里redis存储权限，用在登录过滤器中封装用户权限存储在上下文，在自定义拦截器中需要使用
    /*
    这里继承它用于：指定登录接口，认证、校验成功、校验失败
    此过滤器的作用是用于认证
 */
public class LoginAuthenticationFilter extends UsernamePasswordAuthenticationFilter {

    private final StringRedisTemplate redisTemplate;

    /*构造方法中
    传入AuthenticationManager,它的方法用来校验【不再赘述】
    */
    public LoginAuthenticationFilter(AuthenticationManager manager, StringRedisTemplate redisTemplate) {
        //通过构造赋值
        this.setAuthenticationManager(manager);
        //通过构造赋值
        this.redisTemplate = redisTemplate;
        //是否只处理 POST 请求
        this.setPostOnly(false);
        //只会处理路径为 "/user/login" 且方法为 POST 的请求,即此过滤器只处理登录接口
        this.setRequiresAuthenticationRequestMatcher((new AntPathRequestMatcher("/user/login", "POST")));
    }

    //校验方法
    @Override
    public Authentication attemptAuthentication(HttpServletRequest request, HttpServletResponse response) throws AuthenticationException {
        try {
            //获取传递过来的用户名和密码(将json数据转换为java对象)
            LoginDto loginDto = JSON.parseObject(request.getInputStream(), LoginDto.class);
            //封装Token对象
            Authentication authenticationToken = new UsernamePasswordAuthenticationToken(loginDto.getUsername(), loginDto.getPassword());
            //返回填充
            return this.getAuthenticationManager().authenticate(authenticationToken);
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }

    //认证成功
    @Override
    protected void successfulAuthentication(HttpServletRequest request, HttpServletResponse response, FilterChain chain, Authentication auth) throws IOException, ServletException {
        CustomUser customUser = (CustomUser) auth.getPrincipal();
        /*
            获取当前用户权限数据，放到Redis里面 key：username   value：权限数据
            以Json的格式存储：
            [
                {
                    "authority": " visitor"
                },
                {
                    "authority": "admin"
                },
           ]
         */
        redisTemplate.opsForValue().set(customUser.getUsername(),
                JSON.toJSONString(customUser.getAuthorities()));
        //下发token
        Map<String, Object> map = new HashMap<>();
        map.put("username", customUser.getUsername());
        String token = JwtUtil.creatJwt(map);
        //这里原生方式返回(没有ResponseBody)
        Map<String, Object> returnMap = new HashMap<>();
        returnMap.put("token", token);
        ResponseUtil.out(response, Result.ok(returnMap));
    }

    //认证失败
    @Override
    protected void unsuccessfulAuthentication(HttpServletRequest request, HttpServletResponse response, AuthenticationException failed) throws IOException, ServletException {
        ResponseUtil.out(response, Result.build(201, "认证失败", null));
    }
}

    # 登录校验过滤器

    • 就是普通过滤器，需要在Security中配置了先执行
    • 在这里才存储到上下文，需要用户权限，之前已经存储再来redis
    • 除了登录接口每个接口都要拦截，用过滤器来解决，携带token，将token的信息封装为对象存储到Security上下文(从redis获取权限)，放行后进入Security链，通过上下文判断。
    public class LoginCheckFilter extends OncePerRequestFilter {

    // 定义排除的URI列表
    private final List<String> excludedURIs = Arrays.asList("/user/login");

    private final StringRedisTemplate redisTemplate;

    public LoginCheckFilter(StringRedisTemplate redisTemplate) {
        //通过构造注入
        this.redisTemplate = redisTemplate;
    }

    @Override
    protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain chain) throws ServletException, IOException {

        //登录请求，直接放行
        if (excludedURIs.contains(request.getRequestURI())) {
            //放行
            chain.doFilter(request, response);
            //不再向下执行
            return;
        }

        //其它请求通过token判断是否登录
        String token = request.getHeader("token");
        if (token != null) {
            try {
                //解析成功放行【这个方法已经包含了解析】
                String username = JwtUtil.getUsername(token);
                //通过username从redis获取权限为集合形式SimpleGrantedAuthority
                String authString = redisTemplate.opsForValue().get(username);
                List<Map> maplist = JSON.parseArray(authString, Map.class);
                List<SimpleGrantedAuthority> authList = new ArrayList<>();
                for (Map map : maplist) {
                    //authority名称是由Security决定的
                    String authority = (String) map.get("authority");
                    authList.add(new SimpleGrantedAuthority(authority));
                }
                //存储到上下文，此后不会走Security认证过滤器
                Authentication authenticationToken = new UsernamePasswordAuthenticationToken(username, null, authList);
                SecurityContextHolder.getContext().setAuthentication(authenticationToken);
                //放行
                chain.doFilter(request, response);
            } catch (Exception e) {
                //解析失败
                ResponseUtil.out(response, Result.build(201, "token解析失败", null));
            }
        } else {
            ResponseUtil.out(response, Result.build(201, "缺少token", null));
        }
    }
}

    # Security配置类

    注意启用注解@EnableGlobalMethodSecurity(prePostEnabled = true)用于授权

    //WebSecurityConfigurerAdapter控制Security的行为
@Configuration
@EnableGlobalMethodSecurity(prePostEnabled = true)
public class SecurityConfig extends WebSecurityConfigurerAdapter {

    @Autowired
    private UserDetailsService userDetailsService;

    @Autowired
    private CustomMd5PasswordEncoder passwordEncoder;

    @Autowired
    private RedisTemplate redisTemplate;

    @Bean
    @Override
    protected AuthenticationManager authenticationManager() throws Exception {
        return super.authenticationManager();
    }

    @Override
    protected void configure(HttpSecurity http) throws Exception {
        //关闭csrf [默认开启]
        http.csrf().disable();
        //开启跨域
        http.cors();
        //请求控制
        http.authorizeRequests()
                //不需要认证的路径,可变参数
                .antMatchers("/user/login").permitAll()
                //其它路径需要认证
                .anyRequest().authenticated();
        /*
        1. 让自定义的登录校验过滤器LoginCheckFilter()在UsernamePasswordAuthenticationFilter前执行(即认证前)
        2. 将认证过滤器LoginAuthenticationFilter到加入Security过滤器链
         */
        http.addFilterBefore(new LoginCheckFilter(redisTemplate), UsernamePasswordAuthenticationFilter.class).addFilter(new LoginAuthenticationFilter(authenticationManager(), redisTemplate));
        //禁用session【不禁也行】
        http.sessionManagement().sessionCreationPolicy(SessionCreationPolicy.STATELESS);
    }

    @Override
    protected void configure(AuthenticationManagerBuilder auth) throws Exception {
        // 指定UserDetailService和加密器
        auth.userDetailsService(userDetailsService).passwordEncoder(passwordEncoder);
    }

    /**
     * 配置哪些请求不拦截
     * 排除swagger等相关请求
     * 注意下发路径不能缺少，否则swagger文档异常
     */
    @Override
    public void configure(WebSecurity web) throws Exception {
        web.ignoring().antMatchers(
                "/v2/api-docs",
                "/configuration/ui",
                "/swagger-resources/**",
                "/configuration/security",
                "/swagger-ui.html",
                "/webjars/**",
                "/swagger/**",
                "/doc.html"
        );
    }
}

    # 在controller方法上添加注解

    • 常用PreAuthorize
    • 其中可以使用四种方式，这里使用hasAnyAuthority
    @RestController
@RequestMapping("/user")
public class UserController {
    @Autowired
    private UserService userService;

    @ApiOperation("登录接口")
    @PostMapping("login")
    public Result login(@RequestBody LoginDto loginDto) {
        return Result.ok();
    }

    @ApiOperation("公共资源")
    @PreAuthorize("hasAnyAuthority('user/public')")
    @GetMapping("public")
    public Result getPublic() {
        return Result.build(200, "正在访问公共资源", null);
    }

    @ApiOperation("根据id获取用户")
    @PreAuthorize("hasAnyAuthority('user/getUser')")
    @GetMapping("getUser")
    public Result getUser(Long id) {
        PerUser user = userService.getById(id);
        return Result.ok(user);
    }

    @ApiOperation("获取所有用户")
    @PreAuthorize("hasAnyAuthority('user/getAllUser')")
    @GetMapping("getAllUser")
    public Result getAllUser() {
        List<PerUser> list = userService.list();
        return Result.ok(list);
    }

    @ApiOperation("删除用户")
    @PreAuthorize("hasAnyAuthority('user/deleteUser')")
    @DeleteMapping("deleteUser")
    public Result deleteUser(Long id) {
        userService.removeById(id);
        return Result.ok();
    }
}
