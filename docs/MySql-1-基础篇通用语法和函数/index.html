    # 数据库规范

    # 数据库

    • 小写字母
    • 下划线或横线分隔单词

    # 表名

    • 小写字母
    • 下划线或横线分隔单词
    • 复数形式：使用users而不是user做表名

    # 字段名

    • 小写字母

    • 下划线或横线分隔单词

    • 避免使用关键字做字段名

    # 主键

    • 唯一
    • 自增

    # 索引

    • 避免过多索引：只为常用的查询和连接创建索引

    # Sql分类

    分类 全称 描述
    DDL Data Definition Language 数据定义语言，用于创建、修改和删除数据库对象(存储的实体:表格、视图、索引、存储过程等)
    DML Data Manipulation Language 数据操作语言，用于对数据库中的数据进行操作(INSERT、UPDATE和DELETE)
    DQL Data Query Language 数据查询语言,用于从数据库中检索数据(SELECT)
    DCL Data Control Language 数据控制语言,用于控制数据库用户的访问权限

    # MySql数据类型

    # 数值类型

    整数类型支持使用 UNSIGNED，如 age TINYINT UNSIGNED,即(0,2^8]

    数据类型 描述 例子
    TINYINT 1 字节有符号整数 age TINYINT
    SMALLINT 2 字节有符号整数 quantity SMALLINT
    MEDIUMINT 3 字节有符号整数 population MEDIUMINT
    INT 或 INTEGER 4 字节有符号整数 salary INT
    BIGINT 8 字节有符号整数 total_assets BIGINT
    FLOAT 4 字节单精度浮点数 height FLOAT
    DOUBLE 或 REAL 8 字节双精度浮点数 weight DOUBLE
    DECIMAL 精确小数字描述，取决于精度M(整个数字的长度【包含小数点)和标度D(小数位数) price DECIMAL(10, 2)

    # 字符串类型

    • char(n):适合于定长，如手机号。

    • vachar：适合边长，如用户名，性相对char较低(需要判断长度)

    • 下方所说的存储字符数时英文情况(一字符就是一字节)

    数据类型 描述 例子
    CHAR(n) 固定长度字符串，最多包含 n 个字符 code CHAR(10)
    VARCHAR(n) 可变长度字符串，最多包含 n 个字符 name VARCHAR(50)
    TINYTEXT 微型文本，最大长度 255 字符 description TINYTEXT
    TEXT 文本，最大长度 65,535 字符 content TEXT
    MEDIUMTEXT 中型文本，最大长度 16,777,215 字符 remarks MEDIUMTEXT
    LONGTEXT 长文本，最大长度 4,294,967,295 字符 comments LONGTEXT

    # 时间日期类型

    TIMESTAMP不常用，最大到2038年

    数据类型 描述 例子
    DATE 日期，格式 YYYY-MM-DD birthdate DATE
    TIME 时间，格式 HH:MM:SS start_time TIME
    DATETIME 日期和时间，格式 YYYY-MM-DD HH:MM:SS timestamp DATETIME
    TIMESTAMP 时间戳，与时区有关 last_updated TIMESTAMP
    YEAR 年份，格式 YYYY graduation_year YEAR

    # MySql约束

    保证数据的有效性，完整性，正确性

    约束类型 描述 示例
    PRIMARY KEY 唯一标识表中每一行数据，且不能为 NULL id INT PRIMARY KEY
    UNIQUE 保证列或一组列中的所有值都是唯一的，可以有多个唯一约束 email VARCHAR(50) UNIQUE
    FOREIGN KEY 创建两个表之间的关系，确保外键值在另一个表中存在
    比较少用，级别使用逻辑外键    		 FOREIGN KEY (product_id) REFERENCES products(product_id)
    NOT NULL 保证列中的值不为空 username VARCHAR(30) NOT NULL
    CHECK 保证列中的值符合特定的条件 age INT CHECK (age >= 18)
    DEFAULT 为列提供默认值，如果插入时未提供值，则使用默认值 status VARCHAR(10) DEFAULT 'active'

    # MySql条件

    运算符 描述 示例
    = 等于 WHERE column_name = value
    <> 或 != 不等于 WHERE column_name != value
    < 小于 WHERE column_name < value
    > 大于 WHERE column_name > value
    <= 小于等于 WHERE column_name <= value
    >= 大于等于 WHERE column_name >= value
    BETWEEN 在指定范围内 WHERE column_name BETWEEN value1 AND value2
    LIKE 模糊匹配 WHERE column_name LIKE pattern
    IN 在给定值列表中 WHERE column_name IN (value1, value2, ...)
    IS NULL 为 NULL WHERE column_name IS NULL
    IS NOT NULL 不为 NULL WHERE column_name IS NOT NULL
    AND 逻辑 AND WHERE condition1 AND condition2
    OR 逻辑 OR WHERE condition1 OR condition2
    NOT 逻辑 NOT WHERE NOT condition
    IN 子查询 在子查询的结果集中 WHERE column_name IN (SELECT column_name FROM another_table)

    # MySql函数

    # 字符串函数

    函数 描述 示例
    CONCAT() 连接字符串 SELECT * FROM users WHERE CONCAT('%','三','%');
    UPPER() 和 LOWER() 将字符串转为大写或小写 SELECT UPPER(name) FROM products;
    LENGTH() 返回字符串的长度 SELECT LENGTH(description) FROM books;
    SUBSTRING() 提取子字符串 SELECT SUBSTRING(title, 1, 5) FROM movies;
    TRIM() 去除前后空格 SELECT TRIM(' username ');

    # 数学函数

    函数 描述 示例
    ROUND() 四舍五入 SELECT ROUND(price, 2) FROM products;
    ABS() 返回绝对值 SELECT ABS(balance) FROM accounts;
    CEIL() 和 FLOOR() 向上取整和向下取整 SELECT CEIL(amount) FROM transactions;
    MOD(x,y) 取模 SELECT MOD(10, 4) as result;

    # 日期和时间函数

    函数 描述 示例
    NOW() 返回当前日期和时间 SELECT NOW() AS current_datetime;
    CURDATE() 返回当前日期 SELECT CURDATE() as curDate;
    CURTIME() 返回当前时间 SELECT CURTIME() as curTime;
    YEAR()/MONTH()/DAY() 年月日 SELECT YEAR(now()) as year
    DATE_FORMAT() 格式化日期和时间 SELECT DATE_FORMAT(birth_date, '%Y-%m-%d') FROM users;
    DATEDIFF() 计算日期之间的差异 SELECT DATEDIFF(NOW(), hire_date) AS days_since_hire FROM employees;

    # 聚合函数

    函数 描述 示例
    COUNT() 计算行的数量（可以用于计算某个列的非 NULL 值数量） SELECT COUNT(*) FROM students;
    SUM() 计算数值列的总和 SELECT SUM(score) FROM exam_results;
    AVG() 计算数值列的平均值 SELECT AVG(age) FROM employees;
    MIN() 找出数值列的最小值 SELECT MIN(price) FROM products;
    MAX() 找出数值列的最大值 SELECT MAX(salary) FROM employees;
    GROUP_CONCAT() 将多行值连接成一个字符串，可与 GROUP BY 一起使用 SELECT department, GROUP_CONCAT(employee_name) FROM employee GROUP BY department;

    注意

    • count(*):计算所有行数，包含null
    • count(1):与count(*)相同
    • count(username):计算username字段的非null行数

    # LOOP 和 ITERATE

    LOOP循环，ITRERATE迭代

    CREATE PROCEDURE simpleLoopExample()
BEGIN
  -- 声明并初始化计数器
  DECLARE counter INT DEFAULT 1;
  -- 定义循环标签
  myLoop: LOOP
    -- 执行循环体
    -- 检查是否满足退出循环的条件
    IF counter > 10 THEN
      LEAVE myLoop; -- 跳出循环
    END IF;
    -- 更新计数器
    SET counter = counter + 1;
  END LOOP myLoop;
END;

    CREATE PROCEDURE example_iterate()
BEGIN
    -- 声明并初始化计数器
    DECLARE counter INT DEFAULT 0;
    -- 定义循环标签
    loop_label: LOOP
        -- 更新计数器
        SET counter = counter + 1;

        -- 如果计数器等于 3，则使用 ITERATE 跳过后续代码，进入下一次循环迭代
        IF counter = 3 THEN
            ITERATE loop_label;
        END IF;
        -- 如果计数器等于 5，则通过 LEAVE 跳出循环
        IF counter = 5 THEN
            LEAVE loop_label;
        END IF;
    END LOOP loop_label;
END;
