    # 多表查询概述

    # 笛卡尔积

    • 此种方式查询的结果是两种表的所有数据组合即n∗m，不合理
    • 需要消除无效的数据
    -- table1和table2的两个字段相关联
SELECT * FROM table1,table2

    2

    # 消除笛卡尔积

    -- 需要添加条件关联的字段即可
SELECT * FROM users t1,user_roles t2 WHERE t1.user_id=t2.user_id

    # 内连接

    查询两张表的交集,结果不包含null的项，如下方的结果用户5没有分配角色，并不会显示

    -- 隐式内连接

SELECT * FROM tabl1 [INNER] JOIN table2 ON 条件

    # 外连接

    • 可以查询除null的数据

    • 左外连接(使用较多):查询左表所有数据和两张表的公共部分

    • 右外连接:查询有表所有数据和两张表的公共部分

    SELECT * FROM table1 LEFT [OUTER] JOIN table2 ON 条件

    # 自连接

    当前表于自己连接查询，必需要使用表别名

    -- 查询员工和它的领导信息，领导也在员工表，员工有一个leader_id是上级id
-- 实际这里使用的还是内连接
SELECT t1.name AS employee, t2.name AS leader
FROM emp t1
INNER JOIN emp t2 ON t1.leader_id = t2.emp_id;

-- 或
SELECT t1.name AS employee, t2.name AS leader
FROM emp t1, emp t2
WHERE t1.leader_id = t2.emp_id;

    # 联合查询

    • 多条SQL的结果合在一起，使用UNION [ALL]连接
    • UNION 操作符会去除结果中的重复行，加上ALL则不去重
    • 要求：两条SQL查询的数据和类型要一致，下方查询所有
    • 对于单表相当于DISTINCT+OR
    -- 查询年龄<25，收入>8000的员工
SELECT DISTINCT * FROM emp WHERE age < 25 OR income > 8000

-- 或(自动去重)
SELECT * FROM emp WHERE age < 25 UNION SELECT * FROM emp WHERE income > 8000

    # 标量子查询

    • 子查询的结果只有1行1列

    • 常用于单个数字、日期、字符串等的查询

    -- 根据部门名称查询员工信息
SELECT * FROM emp WHERE dep_id=(SELECT dpt_id FROM dpt WHERE name="销售部")

-- 也可以使用内连接，先找出dep_id相同的数据，然后只保留name为销售部的
SELECT emp.*
FROM emp
JOIN dpt ON emp.dep_id = dpt.dpt_id
WHERE dpt.name = "销售部";

    # 列子查询

    子查询的结果只有1列

    操作符 描述
    IN 在指定的集合范围之内，多选一
    NOT IN 不在指定的集合范围之内
    ANY 子查询返回列表中，有任意一个满足即可
    SOME 与ANY等同，使用SOME的地方都可以使用ANY
    ALL 子查询返回列表的所有值都必须满足
    -- 根据部门名称查询销售部和市场部的所有员工
SELECT * FROM emp WHERE id IN(SELECT id FROM dept WHERE name ='销售部' OR name='市场部')

-- 也可以使用内连接
SELECT emp.*
FROM emp
JOIN dept ON emp.id = dept.id
WHERE dept.name ='销售部' OR dept.name='市场部';

    # 行子查询

    子查询的结果只有1行

    -- 查询与'张三'的薪资和直属领导相同的员工信息
SELECT * FROM emp WHERE (salary,leader_id) =(SELECT salary,leader_id FROM emp WHERE name ='张三')

-- 也可以使用内连接(自连接)
SELECT emp.*
FROM emp
JOIN emp AS t1 ON emp.salary = t1.salary AND emp.leader_id = t1.leader_id
WHERE t1.name = '张三';

    # 表子查询

    子查询的结果多行多列

    -- 查询领导和薪资与张三或李四相同的员工信息
SELECT * FROM emp WHERE (salary,leader_id) IN 
(SELECT salary,leader_id FROM emp WHERE name = '张三' OR name = '李四')

-- 或
SELECT emp.*
FROM emp
JOIN emp t1 ON  emp.salary = t1.salary AND emp.leader_id = t1.leader_id
