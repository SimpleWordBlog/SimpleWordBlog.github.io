    # 自定义Realm

    service可直接注入，我这里有循环依赖问题，先通过构造注入

    @Component
public class MyRealm extends AuthorizingRealm {
  
		@Autowired	
    private UserService userService;

  	@Autowired
    private PermissionService permissionService;

    //自定义授权
    @Override
    protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
      	//principals就是产地过来的用户信息
        String username = principals.getPrimaryPrincipal().toString();
        //查询和封装用户的权限等
        SimpleAuthorizationInfo info = new SimpleAuthorizationInfo();
        List<String> permission = permissionService.getPermissionByUsername(username);
        info.addStringPermissions(permission);
        return info;
    }

    //自定义认证
    @Override
    protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
        //获取传入的用户信息
        String username = token.getPrincipal().toString();
        //数据库查询是否存在用户
        LambdaQueryWrapper<PerUser> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(PerUser::getUsername, username);
        PerUser user = userService.getOne(wrapper);
        if (user == null) {
            throw new UnknownAccountException("用户不存在");
        }
        //返回info(接口)
        AuthenticationInfo info = new SimpleAuthenticationInfo(
                token.getPrincipal(),
                user.getPassword(),
                ByteSource.Util.bytes("shiro"),
                username
        );
        return info;
    }
}

    # ShiroConfig

    @Configuration
public class ShiroConfig {
    private final MyRealm myRealm;

    @Autowired
    public ShiroConfig(MyRealm myRealm) {
        this.myRealm = myRealm;
    }

    // 配置SecurityManager
    @Bean
    public DefaultWebSecurityManager securityManager() {
        // DefaultWebSecurityManager对象
        DefaultWebSecurityManager securityManager = new DefaultWebSecurityManager();
        // 加密对象(加盐信息在myRealm配置),数据库需要存储相同的算法加密的密码
        HashedCredentialsMatcher credentialsMatcher = new HashedCredentialsMatcher();
        credentialsMatcher.setHashAlgorithmName("md5");
        credentialsMatcher.setHashIterations(3);
        myRealm.setCredentialsMatcher(credentialsMatcher);
        //将myRealm配置到SecurityManager
        securityManager.setRealm(myRealm);
        return securityManager;
    }

    // 配置拦截
    @Bean
    public ShiroFilterFactoryBean shiroFilterFactoryBean(DefaultWebSecurityManager securityManager) {
        ShiroFilterFactoryBean factoryBean = new ShiroFilterFactoryBean();
        //设置安全管理器
        factoryBean.setSecurityManager(securityManager);
        //设置拦截
        Map<String, String> map = new HashMap<>();
      	//无需拦截
        map.put("/user/login", "anon");
        map.put("/swagger-ui.html", "anon");
        map.put("/swagger/**", "anon");
        map.put("/webjars/**", "anon");
        map.put("/swagger-resources/**", "anon");
        map.put("/v2/**", "anon");
        map.put("/static/**", "anon");
        //其它需要拦截
        map.put("/**", "authc");
        factoryBean.setFilterChainDefinitionMap(map);
        return factoryBean;
    }
}

    不同的拦截类型对应不同的过滤器

    拦截类型 功能
    anon 允许匿名用户访问，不需要进行身份验证
    authc 要求用户进行身份验证，可以使用表单进行登录
    authcBasic 基本 HTTP 身份验证，适用于无状态的请求
    authcBearer 使用 Bearer Token 进行身份验证
    user 要求用户已经身份验证或者在之前的会话中选择了 "记住我" 功能
    logout 登出用户，清除相关的身份验证信息
    roles 要求用户具有指定的角色
    perms 要求用户具有指定的权限
    ssl 强制使用 SSL 安全连接
    port 强制使用指定的端口
    rest REST 风格的请求处理，支持 HTTP 方法过滤
    jwt 处理 JWT（JSON Web Token）身份验证
    rolesOr 用户具有任一指定角色即可通过
    permsOr 用户具有任一指定权限即可通过

    # Contrller

    再方法上添加注解，这里与Security类似，也支持多种注解

    @RestController
@RequestMapping("/user")
public class UserController {
    @Autowired
    private UserService userService;

    @ApiOperation("登录接口")
    @PostMapping("login")
    public Result login(@RequestBody LoginDto loginDto) {
        //获取subject对象
        Subject subject = SecurityUtils.getSubject();
        //封装请求对象到认证token
        AuthenticationToken authToken = new UsernamePasswordToken(loginDto.getUsername(), loginDto.getPassword());
        //认证
        try {
            subject.login(authToken);
            return Result.ok();
        } catch (AuthenticationException e) {
            throw new BaseException(201, "登录失败");
        }
    }


    @ApiOperation("公共资源")
    @GetMapping("public")
    public Result getPublic() {
        return Result.build(200, "正在访问公共资源", null);
    }

    @RequiresPermissions("user/getUser")
    @ApiOperation("根据id获取用户")
    @GetMapping("getUser")
    public Result getUser(Long id) {
        PerUser user = userService.getById(id);
        return Result.ok(user);
    }

    @RequiresPermissions("user/getAllUser")
    @ApiOperation("获取所有用户")
    @GetMapping("getAllUser")
    public Result getAllUser() {
        List<PerUser> list = userService.list();
        return Result.ok(list);
    }

    @RequiresPermissions("user/deleteUser")
    @ApiOperation("删除用户")
    @DeleteMapping("deleteUser")
    public Result deleteUser(Long id) {
        userService.removeById(id);
        return Result.ok();
    }
}

    # 退出

    之前的配置不变，subject就是用户，这里可以清除redis等操作

    @ApiOperation("退出")
@GetMapping("logout")
public Result logout()
{
    Subject subject = SecurityUtils.getSubject();
    if(subject.getPrincipal() != null)
    {
        subject.logout();
        return Result.ok();
    }
    return Result.fail();
}
