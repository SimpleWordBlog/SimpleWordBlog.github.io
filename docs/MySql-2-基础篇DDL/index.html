    通过`SHOW CREATE TABLE students`查看结构

    • ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 存储引擎和字符集
    • 这里使用了当前数据库默认的，也可以在建表时在最后自定义添加
    CREATE TABLE `students` (
  `student_id` int(11) NOT NULL COMMENT '学生ID，主键',
  `first_name` varchar(50) NOT NULL COMMENT '姓',
  `last_name` varchar(50) NOT NULL COMMENT '名',
  `birth_date` date DEFAULT NULL COMMENT '出生日期',
  `enrollment_date` date DEFAULT NULL COMMENT '入学日期',
  `major` varchar(50) DEFAULT NULL COMMENT '专业',
  `GPA` float DEFAULT NULL COMMENT '平均绩点',
  PRIMARY KEY (`student_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='学生信息表'

    # 表修改

    # 添加字段

    ALTER TABLE 表名 ADD 字段名 data_type [约束] [注释] ;
    1
    ALTER TABLE students ADD email VARCHAR(100) UNIQUE NOT NULL COMMENT '电子邮件唯一且不能为空';
    1

    # 修改字段数据类型

    ALTER TABLE 表名 MODIFY 字段名 数据类型
    1
    ALTER TABLE students MODIFY email char(100)
    1

    # 修改字段名和字段类型

    ALTER TABLE 表名 CHANGE 旧字段 新字段 新字段数据类型 [约束] [注释]
    1
    ALTER TABLE students CHANGE email stu_email varchar(100) COMMENT '修改字段名为和字段类型'
    1

    # 修改表名

    ALTER TABLE students RENAME TO student2
    1

    # 表删除

    # 删除表

    DROP TABLE [IF EXISTS] 表名
    1

    # 删除并重建该表(空表)

    TRUNCATE TABLE 表名
