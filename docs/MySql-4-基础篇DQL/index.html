    MySql-DQL数据查询

    是秃子总会发光 Mysql

    # 基础查询

    # 查询字段

    提示

    SELECT *效率较低，可读性差

    SELECT user_id, username, password, email FROM users
    1
    SELECT * FROM users
    1

    # 别名AS

    • 使用：字段或表 AS 别名
    • AS可以省略
    SELECT u.user_id, u.username, u.password, u.email FROM users AS u
    1

    这样查询的结果表字段将会以别名展示

    SELECT user_id id, username name FROM users
    1

    # 数据去重DISTINCT

    对查询的结果去重

    SELECT DISTINCT email FROM users
    1

    # 条件查询

    更对条件见基础知识

    # BETWEN…AND多条件

    -- 年龄在[18,25]且成绩不小于80的数据
SELECT * FROM students WHERE age BETWEEN 18 AND 25 AND grade >= 80
    1
    2

    # 模糊查询

    # %

    %表示匹配任意(个)字符

    这里也可以使用CONCAT()函数替代

    -- 查询名字以J开头的学生，
SELECT * FROM students WHERE name LIKE 'J%'
    1
    2
    -- 查询名字包含an的学生
SELECT * FROM students WHERE name LIKE '%an%';
    1
    2
    -- 查询QQ邮箱的学生
SELECT * FROM students WHERE email LIKE '%@qq.com';
    1
    2

    # __

    • _匹配任意一个字符
    • 可以使用多个,匹配多个字符
    -- 查询名字第三个字符是a的学生
SELECT * FROM students WHERE name LIKE '__a%';
    1
    2

    # IN子查询(替换OR)

    SELECT * FROM students WHERE student_id IN (101, 102, 105);
    1
    SELECT * FROM students WHERE grade IN (SELECT grade FROM exam_results WHERE subject = 'Math' AND score >= 80);
    1

    # 聚合查询

    • 参考聚合函数
    • 可以在SELECT后和HAVING条件使用
    聚合函数 描述 示例
    COUNT() 计算行的数量（可以用于计算某个列的非 NULL 值数量）;
    *代表所有字段，即所有数据    		 SELECT COUNT(*) FROM students;
    SUM() 计算数值列的总和 SELECT SUM(score) FROM exam_results;
    AVG() 计算数值列的平均值 SELECT AVG(age) FROM employees;
    MIN() 找出数值列的最小值 SELECT MIN(price) FROM products;
    MAX() 找出数值列的最大值 SELECT MAX(salary) FROM employees;
    GROUP_CONCAT() 将多行值连接成一个字符串，可与 GROUP BY 一起使用 SELECT department, GROUP_CONCAT(employee_name) FROM employee GROUP BY department;

    # 分组查询

    • where和having都是条件，where作用于分组前，having作用于分组
    • where条件不能使用聚合函数，having可以
    SELECT * FROM table_name [WHERE condition] GROUP BY column1,column2 [HAVING condition]
    1
    -- 查询薪水>50000部门和人数,要求按部门分组，过滤掉人数小于1的部门
SELECT department, COUNT(*) AS employee_count
FROM employees
WHERE salary > 50000
GROUP BY department
HAVING COUNT(*) > 1;
    1
    2
    3
    4
    5
    6

    # 排序查询

    • 排序在查询结果之后，即SELECT之后执行，在分页前执行

    • ASC可省略，即默认升序

    SELECT * FROM table_name [WHERE condition] ORDER BY column1 [ASC | DESC],column2 [ASC | DESC],...
    1
    -- 查询it部门的员工，按薪水降序排序
SELECT * FROM employees WHERE department = 'IT' ORDER BY salary DESC;
    1
    2
    -- 查询产品，先按价格降序，相同则按照质量分排序
SELECT * FROM products ORDER BY price DESC,quantity_sold DESC;
    1
    2

    # 分页查询

    • mysql默认排序不稳定，可能出现分页是字段缺少等问题
    • 推荐每次分页都自定义排序
    SELECT * FROM table_name [ORDER BY] LIMIT 起始索引，查询记录数
    1

    pageCurrent(查询页) pageSize(每页数目) 查询(类似数组)
    1 5 0-5
    2 5 5-10
    3 5 10-15

    起始索引=(pageCurrent-1)*pageSize

    即

    SELECT * FROM table_name [ORDER BY] LIMIT pageCurrent-1)*pageSize，pageSize
    1
    -- 第一页 (查询第一页时，第一个参数0可以省略，必然是0)
SELECT * FROM users LIMIT 0,2

-- 第三页(升序查询)
SELECT * FROM users ORDER BY user_id LIMIT 4,2;
    1
    2
    3
    4
    5

    # DQL的编写和执行顺序

    # 编写顺序

    SELECT>FROM>WHERE>GROUP BY>HAVING>ORDER BY>LIMIT
    1

    执行顺序

    FROM>WHERE>GROUP BY>HAVING>SELECT>ORDER BY>LIMIT
    1
