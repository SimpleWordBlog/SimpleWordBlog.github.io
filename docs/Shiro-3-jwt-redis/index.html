    # JWTToken

    • 之前表单认证，我们使用的是UsernamePasswordToken，是AuthenticationToken的子类
    • 现在需要使用jwt认证，我们自行实现接口
    public class JWTToken implements AuthenticationToken {

    private String token;

    public JWTToken(String token) {
        this.token = token;
    }

    @Override
    public Object getPrincipal() {
      	//之前这里返回的是用户名
        return token;
    }

    @Override
    public Object getCredentials() {
      	//之前这里返回的是输入的密码
        return token;
    }
}

    # 自定义过滤器

    • 这里由多种方式，我使用普通过滤器
    • shiro提供的过滤器(常用两个，自行网络)
    • 在配置类中添加对它的支持，参考Security
    public class JwtFilter extends OncePerRequestFilter {

    // 定义排除的URI列表
    private final List<String> excludedURIs = Arrays.asList("/user/login");

    @Override
    protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain chain) throws ServletException, IOException {

        //登录请求，直接放行
        if (excludedURIs.contains(request.getRequestURI())) {
            //放行
            chain.doFilter(request, response);
            //不再向下执行
            return;
        }

        //其它请求通过token判断是否登录
        String token = request.getHeader("token");
        if (token != null) {
            try {
                //token有效，开启认证
                SecurityUtils.getSubject().login(new JWTToken(token));
                chain.doFilter(request, response);
            } catch (Exception e) {
                //解析失败
                ResponseUtil.out(response, Result.build(201, "token解析失败", null));
            }
        } else {
            ResponseUtil.out(response, Result.build(201, "缺少token", null));
        }
    }
}

    # 自定义Realm

    @Component
public class MyRealm extends AuthorizingRealm {

    @Autowired
    private UserService userService;

    @Autowired
    private PermissionService permissionService;

   //需要添加对JWTToken(自定义的认证token)的支持，否则jwt解析报错
    @Override
    public boolean supports(AuthenticationToken token) {
        return token instanceof JWTToken;
    }

    //自定义授权
    @Override
    protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
      	//现在传递过来的就是jwt了
        String jwt = principals.toString();
        String username = JwtUtil.getUsername(jwt);
        //查询和封装用户的权限等(这些内容可以使用redis，这里不再演示)
        SimpleAuthorizationInfo info = new SimpleAuthorizationInfo();
        List<String> permission = permissionService.getPermissionByUsername(username);
        info.addStringPermissions(permission);
        return info;
    }

    //自定义认证
    @Override
    protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authenticationToken) throws AuthenticationException {
        //获取传入的token
        String token = authenticationToken.getPrincipal().toString();
        //数据库查询是否存在用户
        String username = JwtUtil.getUsername(token);
        LambdaQueryWrapper<PerUser> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(PerUser::getUsername, username);
        PerUser user = userService.getOne(wrapper);
        if (user == null) {
            throw new RuntimeException("用户不存在");
        }
        //返回info(接口)
        AuthenticationInfo info = new SimpleAuthenticationInfo(
                token,
                token,
                username
        );
        return info;
    }
}

    # ShiroConfig

    这里还可以在安全管理器配置session、remberMe等，参考SEcurity

    @Configuration
public class ShiroConfig {
    private final MyRealm myRealm;

    @Autowired
    public ShiroConfig(MyRealm myRealm) {
        this.myRealm = myRealm;
    }

    // 配置SecurityManager，这里使用jwt完成认证，不是用户名和密码认证不再需要加密
    @Bean
    public DefaultWebSecurityManager securityManager() {
        //DefaultWebSecurityManager对象
        DefaultWebSecurityManager securityManager = new DefaultWebSecurityManager();
        //将myRealm配置到SecurityManager
        securityManager.setRealm(myRealm);
        return securityManager;
    }

    // 配置拦截
    @Bean
    public ShiroFilterFactoryBean shiroFilterFactoryBean(DefaultWebSecurityManager securityManager) {
        ShiroFilterFactoryBean factoryBean = new ShiroFilterFactoryBean();
        //设置安全管理器
        factoryBean.setSecurityManager(securityManager);
        //配置自定义过滤器JwtFilter，jwt-JwtFilter
        HashMap<String, Filter> filterMap = new HashMap<>();
        filterMap.put("jwt", new JwtFilter());
        factoryBean.setFilters(filterMap);
        //设置拦截
        Map<String, String> map = new HashMap<>();
        map.put("/user/login", "anon");
        //其它需要拦截，根据上方配置。"jwt"对应走JwtFilter拦截器
        map.put("/**", "jwt");
        factoryBean.setFilterChainDefinitionMap(map);
        return factoryBean;
    }
}

    # controller

    • controoler的登录接口变动
    • 登录接口通过表单认证没有Jwt，不能使用当前的jwt认证
      • 只需要校验用户名和密码即可
      • 下发token
    @ApiOperation("登录接口")
@PostMapping("login")
public Result login(@RequestBody LoginDto loginDto)
{
    //确认用户
    PerUser user = userService.getUserByUsername(loginDto.getUsername());
    if(user == null)
    {
        throw new UserNotFindException(201, "用户不存在");
    }
    //确认密码
    Md5Hash md5Password = new Md5Hash(loginDto.getPassword(), "shiro", 3);
    if(!user.getPassword().equals(md5Password.toString()))
    {
        throw new BaseException(201, "密码不正确");
    }
    //下发jwt令牌
    Map < String, Object > map = new HashMap < > ();
    map.put("username", loginDto.getUsername());
    String jwt = JwtUtil.creatJwt(map);
    return Result.ok(jwt);
}
